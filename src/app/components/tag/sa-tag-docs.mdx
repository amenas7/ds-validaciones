### ‚ö° TypeScript Component

```typescript
import { Component } from '@angular/core';

@Component({
  ......
})

export class ........ {
// Propiedades din√°micas para el ejemplo
  tagText: string = 'Angular';
  tagType: 'primary' | 'secondary' | 'success' | 'danger' | 'warning' | 'info' | 'dark' | 'light' = 'success';
  tagSize: 'small' | 'medium' | 'large' = 'medium';
  customClass: string = '';

  // Arrays para rotaci√≥n de valores
  private texts: string[] = ['Angular', 'TypeScript', 'JavaScript', 'Activo', 'Pendiente', 'Alta Prioridad', 'Frontend', 'Backend'];
  private types: ('primary' | 'secondary' | 'success' | 'danger' | 'warning' | 'info' | 'dark' | 'light')[] = [
    'primary', 'secondary', 'success', 'danger', 'warning', 'info', 'dark', 'light'
  ];
  private sizes: ('small' | 'medium' | 'large')[] = ['small', 'medium', 'large'];

  private currentTextIndex: number = 0;
  private currentTypeIndex: number = 2; // Inicia en 'success'
  private currentSizeIndex: number = 1; // Inicia en 'medium'

  // M√©todos para cambiar propiedades din√°micamente
  changeText(): void {
    this.currentTextIndex = (this.currentTextIndex + 1) % this.texts.length;
    this.tagText = this.texts[this.currentTextIndex];
  }

  changeType(): void {
    this.currentTypeIndex = (this.currentTypeIndex + 1) % this.types.length;
    this.tagType = this.types[this.currentTypeIndex];
  }

  changeSize(): void {
    this.currentSizeIndex = (this.currentSizeIndex + 1) % this.sizes.length;
    this.tagSize = this.sizes[this.currentSizeIndex];
  }
}
```
