### ‚ö° TypeScript Component

```typescript
import { Component } from '@angular/core';

@Component({
  ......
})

export class ........ {
// Propiedades din√°micas para el ejemplo
  legendColor: string = '#28a745';
  legendTooltip: string = 'Este es un tooltip informativo que muestra detalles adicionales';
  tooltipPosition: 'top' | 'bottom' | 'left' | 'right' = 'top';
  customClass: string = '';

  // Arrays para rotaci√≥n de valores
  private colors: string[] = ['#28a745', '#ffc107', '#dc3545', '#007bff', '#6f42c1', '#17a2b8', '#e83e8c', '#20c997'];
  private tooltips: string[] = [
    'Este es un tooltip informativo que muestra detalles adicionales',
    'Leyenda de ventas completadas - $543,210 en ingresos',
    'Productos en stock: 1,234 unidades disponibles',
    'Usuarios activos en los √∫ltimos 30 d√≠as',
    'Tooltip largo: Esta leyenda representa informaci√≥n estad√≠stica importante que requiere una explicaci√≥n m√°s detallada para el usuario',
    'Estado: Todo funcionando correctamente'
  ];
  private positions: ('top' | 'bottom' | 'left' | 'right')[] = ['top', 'bottom', 'left', 'right'];

  private currentColorIndex: number = 0;
  private currentTooltipIndex: number = 0;
  private currentPositionIndex: number = 0;

  // M√©todos para cambiar propiedades din√°micamente
  changeColor(): void {
    this.currentColorIndex = (this.currentColorIndex + 1) % this.colors.length;
    this.legendColor = this.colors[this.currentColorIndex];
  }

  changeTooltip(): void {
    this.currentTooltipIndex = (this.currentTooltipIndex + 1) % this.tooltips.length;
    this.legendTooltip = this.tooltips[this.currentTooltipIndex];
  }

  changeTooltipPosition(): void {
    this.currentPositionIndex = (this.currentPositionIndex + 1) % this.positions.length;
    this.tooltipPosition = this.positions[this.currentPositionIndex];
  }
}
```
